# CV.net Master Schema - Version 2.0

# 1. Enums for Data Integrity
enum EmploymentStatus {
  Employed
  Unemployed
}

enum SkillLevel {
  Beginner
  Intermediate
  Expert
}

enum LanguageLevel {
  Beginner
  Intermediate
  Expert
}

# 2. Core User Table
type User @table {
  id: UUID! @default(expr: "uuid_generate_v4()")
  fullName: String!
  email: String! @unique
  phone: String
  address: String
  portfolioUrl: String
  employmentStatus: EmploymentStatus!
  currentOrg: String # Nullable for Unemployed
  currentPosition: String # Nullable for Unemployed
  personalStatement: String # Professional elevator pitch
  aboutMe: String # Personal passions and narrative bio
  createdAt: Timestamp! @default(expr: "request.time")
  updatedAt: Timestamp! @default(expr: "request.time")
}

# 3. Relational Sub-Tables
